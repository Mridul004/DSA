class Solution {
public:
    int canCompleteCircuit(vector<int>& gas, vector<int>& cost) {
        int carry=0,pos=-1,n=gas.size();
        for(int i=n-1;i>=0;i--){
            int k=gas[i]-cost[i];
            carry+=k;
        }
    }
            if(carry>0&&k>=0) pos=i;
        if(carry>=0) return pos;
        return -1;
};
